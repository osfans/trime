103  private class PopupTimer extends Handler implements Runnable {
104    private int mParentLocation[] = new int[2];
105
106    void postShowFloatingWindow() {
107      if (Function.isEmpty(Rime.getCompositionText())) {
108        hideComposition();
109        return;
110      }
111      mCompositionContainer.measure(LayoutParams.WRAP_CONTENT,
112              LayoutParams.WRAP_CONTENT);
113      mFloatingWindow.setWidth(mCompositionContainer.getMeasuredWidth());
114      mFloatingWindow.setHeight(mCompositionContainer.getMeasuredHeight());
115      post(this);
116    }
117
118    void cancelShowing() {
119      if (null != mFloatingWindow && mFloatingWindow.isShowing()) mFloatingWindow.dismiss();
120      removeCallbacks(this);
121    }
122
123    public void run() {
124      if (mCandidateContainer == null || mCandidateContainer.getWindowToken() == null) return;
125      int x, y;
126      if (isWinFixed() || !cursorUpdated) {
127        //setCandidatesViewShown(true);
128        switch (winPos) {
129          case TOP_RIGHT:
130            mCandidateContainer.getLocationOnScreen(mParentLocation);
131            x = mCandidateContainer.getWidth() - mFloatingWindow.getWidth();
132            y = - mParentLocation[1] + candSpacing;
133            break;
134          case TOP_LEFT:
135            mCandidateContainer.getLocationOnScreen(mParentLocation);
136            x = 0;
137            y = - mParentLocation[1] + candSpacing;
138            break;
139          case BOTTOM_RIGHT:
140            mCandidateContainer.getLocationInWindow(mParentLocation);
141            x = mCandidateContainer.getWidth() - mFloatingWindow.getWidth();
142            y = mParentLocation[1] - mFloatingWindow.getHeight() - candSpacing;
143            break;
144          case DRAG:
145            x = winX;
146            y = winY;
147            break;
148          case FIXED:
149          case BOTTOM_LEFT:
150          default:
151            mCandidateContainer.getLocationInWindow(mParentLocation);
152            x = mParentLocation[0];
153            y = mParentLocation[1] - mFloatingWindow.getHeight() - candSpacing;
154            break;
155        }
156      } else {
157        //setCandidatesViewShown(false);
158        mCandidateContainer.getLocationOnScreen(mParentLocation);
159        x = (int)mPopupRectF.left;
160        if (x > mCandidateContainer.getWidth() - mFloatingWindow.getWidth()) {
161          x = mCandidateContainer.getWidth() - mFloatingWindow.getWidth();
162        } else if (x < 0) x = 0;
163        y = (int)mPopupRectF.bottom - mParentLocation[1] + candSpacing;
164        if (y + mFloatingWindow.getHeight() > 0 || winPos == WinPos.LEFT_UP || winPos == WinPos.RIGHT_UP) {
165          y = (int)mPopupRectF.top - mParentLocation[1] -  mFloatingWindow.getHeight() - candSpacing;
166          if (y < - mParentLocation[1]) y = - mParentLocation[1];
167        }
168      }
169      if (!mFloatingWindow.isShowing()) {
170        mFloatingWindow.showAtLocation(mCandidateContainer,
171                Gravity.START | Gravity.TOP, x, y);
172      } else {
173        mFloatingWindow.update(x, y,
174                mFloatingWindow.getWidth(),
175                mFloatingWindow.getHeight());
176      }
177    }
178  }

210    String[] ss;
211    s = mConfig.getString("locale");
212    if (Function.isEmpty(s)) s = "";
213    ss = s.split("[-_]");
214    if (ss.length == 2) locales[0] = new Locale(ss[0], ss[1]);
215    else if (ss.length == 3) locales[0] = new Locale(ss[0], ss[1], ss[2]);
216    else locales[0] = Locale.getDefault();
217    s = mConfig.getString("latin_locale");
218    if (Function.isEmpty(s)) s = "en_US";
219    ss = s.split("[-_]");
220    if (ss.length == 1) locales[1] = new Locale(ss[0]);
221    else if (ss.length == 2) locales[1] = new Locale(ss[0], ss[1]);
222    else if (ss.length == 3) locales[1] = new Locale(ss[0], ss[1], ss[2]);
223    else locales[0] = Locale.ENGLISH;
224
225    orientation = getResources().getConfiguration().orientation;
226    // Use the following line to debug IME service.
227    //android.os.Debug.waitForDebugger();
228  }

252  private void loadBackground() {
253      GradientDrawable gd = new GradientDrawable();
254      gd.setStroke(mConfig.getPixel("layout/border"), mConfig.getColor("border_color"));
255      gd.setCornerRadius(mConfig.getFloat("layout/round_corner"));
256      Drawable d = mConfig.getDrawable("layout/background");
257      if (d == null) {
258        gd.setColor(mConfig.getColor("text_back_color"));
259        d = gd;
260      }
261      if (mConfig.hasKey("layout/alpha")) {
262        int alpha = mConfig.getInt("layout/alpha");
263        if (alpha <= 0) alpha = 0;
264        else if (alpha >= 255) alpha = 255;
265        d.setAlpha(alpha);
266      }
267      mFloatingWindow.setBackgroundDrawable(d);
268      if (VERSION.SDK_INT >= VERSION_CODES.LOLLIPOP) mFloatingWindow.setElevation(mConfig.getPixel("layout/elevation"));
269      mCandidateContainer.setBackgroundColor(mConfig.getColor("back_color"));

323  @Override
324  public void onUpdateCursorAnchorInfo(CursorAnchorInfo cursorAnchorInfo) {
325    if (VERSION.SDK_INT >= VERSION_CODES.LOLLIPOP) {
326      int i = cursorAnchorInfo.getComposingTextStart();
327      if ((winPos == WinPos.LEFT || winPos == WinPos.LEFT_UP) && i >= 0 ) {
328        mPopupRectF = cursorAnchorInfo.getCharacterBounds(i);
329      } else {
330        mPopupRectF.left = cursorAnchorInfo.getInsertionMarkerHorizontal();
331        mPopupRectF.top = cursorAnchorInfo.getInsertionMarkerTop();
332        mPopupRectF.right =  mPopupRectF.left;
333        mPopupRectF.bottom = cursorAnchorInfo.getInsertionMarkerBottom();
334      }
335      cursorAnchorInfo.getMatrix().mapRect(mPopupRectF);
336      if (mCandidateContainer != null) {
337        mFloatingWindowTimer.postShowFloatingWindow();
338      }
339    }
340  }

342  @Override
343  public void onUpdateSelection(int oldSelStart, int oldSelEnd, int newSelStart,
344      int newSelEnd, int candidatesStart, int candidatesEnd) {
345    super.onUpdateSelection(oldSelStart, oldSelEnd, newSelStart, newSelEnd, 
346        candidatesStart, candidatesEnd);
347    if ((candidatesEnd != -1) &&
348        ((newSelStart != candidatesEnd) || (newSelEnd != candidatesEnd))) {
349      //移動光標時，更新候選區
350      if ((newSelEnd < candidatesEnd) && (newSelEnd >= candidatesStart)) {
351        int n = newSelEnd - candidatesStart;
352        Rime.RimeSetCaretPos(n);
353        updateComposing();
354      }
355    }
356    if ((candidatesStart == -1 && candidatesEnd == -1) &&
357        (newSelStart == 0 && newSelEnd == 0)) {
358      //上屏後，清除候選區
359      escape();
360    }
361    // Update the caps-lock status for the current cursor position.
362    updateCursorCapsToInputView();
363  }

398  /**
399   * 重置鍵盤、候選條、狀態欄等，進入文本框時通常會調用。
400   * @param attribute 文本框的{@link EditorInfo 屬性}
401   * @param restarting 是否重啓
402   */
403  @Override
404  public void onStartInput(EditorInfo attribute, boolean restarting) {
405    super.onStartInput(attribute, restarting);
406    canCompose = false;
407    enterAsLineBreak = false;
408    mTempAsciiMode = false;
409    int inputType = attribute.inputType;
410    int inputClass = inputType & InputType.TYPE_MASK_CLASS;
411    int variation = inputType & InputType.TYPE_MASK_VARIATION;
412    String keyboard = null;
413    switch (inputClass) {
414      case InputType.TYPE_CLASS_NUMBER:
415      case InputType.TYPE_CLASS_PHONE:
416      case InputType.TYPE_CLASS_DATETIME:
417        mTempAsciiMode = true;
418        keyboard = "number";
419        break;
420      case InputType.TYPE_CLASS_TEXT:
421      default:
422        if (variation == InputType.TYPE_TEXT_VARIATION_SHORT_MESSAGE) {
423          // Make enter-key as line-breaks for messaging.
424          enterAsLineBreak = true;
425        }
426        if (variation == InputType.TYPE_TEXT_VARIATION_EMAIL_ADDRESS
427         || variation == InputType.TYPE_TEXT_VARIATION_PASSWORD
428         || variation == InputType.TYPE_TEXT_VARIATION_VISIBLE_PASSWORD
429         || variation == InputType.TYPE_TEXT_VARIATION_WEB_EMAIL_ADDRESS
430         || variation == InputType.TYPE_TEXT_VARIATION_WEB_PASSWORD
431         ) {
432           mTempAsciiMode = true;
433        } else {
434          canCompose = true;
435        }
436        break;
437    }
438    Rime.get();
439    if (reset_ascii_mode) mAsciiMode = false;
440    // Select a keyboard based on the input type of the editing field.
441    mKeyboardSwitch.init(getMaxWidth()); //橫豎屏切換時重置鍵盤
442    mKeyboardSwitch.setKeyboard(keyboard); //設定默認鍵盤
443    updateAsciiMode();
444    if (!onEvaluateInputViewShown()) setCandidatesViewShown(canCompose && !Rime.isEmpty()); //實體鍵盤
445    if (display_tray_icon) showStatusIcon(R.drawable.status); //狀態欄圖標
446  }

484  //句首自動大小寫
485  private void updateCursorCapsToInputView() {
486    if (auto_caps.contentEquals("false") || Function.isEmpty(auto_caps)) return;
487    if ((auto_caps.contentEquals("true") || Rime.isAsciiMode()) && (mKeyboardView != null && !mKeyboardView.isCapsOn())) {
488      InputConnection ic = getCurrentInputConnection();
489      if (ic != null) {
490        int caps = 0;
491        EditorInfo ei = getCurrentInputEditorInfo();
492        if ((ei != null) && (ei.inputType != EditorInfo.TYPE_NULL)) {
493          caps = ic.getCursorCapsMode(ei.inputType);
494        }
495        mKeyboardView.setShifted(false, caps != 0);
496      }
497    }
498  }

538  public boolean handleAciton(int code, int mask) { //編輯操作
539    InputConnection ic = getCurrentInputConnection();
540    if (ic == null) return false;
541    if (code == KeyEvent.KEYCODE_CLEAR) {
542      ic.beginBatchEdit();
543      ic.performContextMenuAction(android.R.id.selectAll);
544      ic.performContextMenuAction(android.R.id.cut);
545      ic.endBatchEdit();
546      return true;
547    }
548    if (VERSION.SDK_INT < VERSION_CODES.HONEYCOMB) return false;
549    if (KeyEvent.metaStateHasModifiers(mask, KeyEvent.META_CTRL_ON)) {
550      // android.R.id. + selectAll, startSelectingText, stopSelectingText, cut, copy, paste, copyUrl, or switchInputMethod
551      if (code == KeyEvent.KEYCODE_A)
552        return ic.performContextMenuAction(android.R.id.selectAll);
553      if (code == KeyEvent.KEYCODE_X)
554        return ic.performContextMenuAction(android.R.id.cut);
555      if (code == KeyEvent.KEYCODE_C)
556        return ic.performContextMenuAction(android.R.id.copy);
557      if (code == KeyEvent.KEYCODE_V)
558        return ic.performContextMenuAction(android.R.id.paste);
559    }
560    return false; 
561  }

593  /** 處理實體鍵盤事件
594   * @param event {@link KeyEvent 按鍵事件}
595   * @return 是否成功處理
596   * */
597  private boolean onKeyEvent(KeyEvent event) {
598    Log.info("onKeyEvent="+event);
599    keyComposing = false;
600    int keyCode = event.getKeyCode();
601    if (keyCode == KeyEvent.KEYCODE_VOLUME_DOWN ||
602        keyCode == KeyEvent.KEYCODE_VOLUME_UP ||
603        keyCode == KeyEvent.KEYCODE_VOLUME_MUTE ||
604        keyCode == KeyEvent.KEYCODE_HOME ||
605        keyCode == KeyEvent.KEYCODE_MENU ||
606        keyCode == KeyEvent.KEYCODE_SEARCH) return false;
607    keyComposing = isComposing();
608    if (!keyComposing) {
609      if (keyCode == KeyEvent.KEYCODE_DEL ||
610          keyCode == KeyEvent.KEYCODE_ENTER ||
611          keyCode == KeyEvent.KEYCODE_BACK) {
612        return false;
613      }
614    } else if (keyCode == KeyEvent.KEYCODE_BACK) {
615      keyCode = KeyEvent.KEYCODE_ESCAPE; //返回鍵清屏
616    }
617
618    if (VERSION.SDK_INT >= VERSION_CODES.HONEYCOMB) {
619      if (KeyEvent.KEYCODE_SPACE == keyCode && event.isCtrlPressed())
620        return handleOption(KeyEvent.KEYCODE_MENU); //切換輸入法
621    }
622
623    int c = event.getUnicodeChar();
624    if (c > 0x20) { //實體鍵盤空格、回車等發送onKey，字符發送onText
625      onText(String.valueOf((char)c));
626      return true;
627    }
628    onKey(keyCode, event.getMetaState());
629    return true;
630  }

632  private IBinder getToken() {
633      final Dialog dialog = getWindow();
634      if (dialog == null) {
635          return null;
636      }
637      final Window window = dialog.getWindow();
638      if (window == null) {
639          return null;
640      }
641      return window.getAttributes().token;
642  }

644  public void onEvent(Event event) {
645    String s = event.getText();
646    if (!Function.isEmpty(s)) {
647      onText(s);
648    } else if (event.code > 0) {
649      int code = event.code;
650      if (code == KeyEvent.KEYCODE_SWITCH_CHARSET) { //切換狀態
651        Rime.toggleOption(event.getToggle());
652        commitText();
653        updateComposing();
654      } else if (code == KeyEvent.KEYCODE_EISU) { //切換鍵盤
655        mKeyboardSwitch.setKeyboard(event.select);
656        //根據鍵盤設定中英文狀態，不能放在Rime.onMessage中做
657        mTempAsciiMode = mKeyboardSwitch.getAsciiMode(); //切換到西文鍵盤時不保存狀態
658        updateAsciiMode();
659        bindKeyboardToInputView();
660        updateComposing();
661      } else if (code == KeyEvent.KEYCODE_LANGUAGE_SWITCH) { //切換輸入法
662        IBinder imeToken = getToken();
663        InputMethodManager imm = (InputMethodManager)getSystemService(INPUT_METHOD_SERVICE);
664        if (event.select.contentEquals(".next") && VERSION.SDK_INT >= VERSION_CODES.JELLY_BEAN) {
665          imm.switchToNextInputMethod(imeToken, false);
666        } else {
667          imm.switchToLastInputMethod(imeToken);
668        }
669      } else if (code == KeyEvent.KEYCODE_FUNCTION) { //命令直通車
670        s = Function.handle(this, event.command, event.option);
671        if (s != null) {
672          commitText(s);
673          updateComposing();
674        }
675      } else if (code == KeyEvent.KEYCODE_VOICE_ASSIST) { //語音輸入
676        new Speech(this).start();
677      } else if (code == KeyEvent.KEYCODE_SETTINGS) { //設定
678        switch (event.option) {
679          case "theme":
680            showThemeDialog();
681            break;
682          case "color":
683            showColorDialog();
684            break;
685          case "schema":
686            showSchemaDialog();
687            break;
688          default:
689            Function.showPrefDialog(this);
690            break;
691        }
692      } else if (code == KeyEvent.KEYCODE_PROG_RED) { //配色方案
693        showColorDialog();
694      } else {
695        onKey(event.code, event.mask);
696      }
697    }
698  }

700  public void onKey(int primaryCode, int mask) { //軟鍵盤
701    if (handleOption(primaryCode)) {
702      Log.info("onOption");
703    } else if (Rime.onKey(Event.getRimeEvent(primaryCode, mask))) {
704      Log.info("Rime onKey");
705      commitText();
706      updateComposing();
707    } else if (handleEnter(primaryCode)
708      || handleAciton(primaryCode, mask)
709      || handleBack(primaryCode)) {
710      Log.info("Trime onKey");
711    } else if (VERSION.SDK_INT >= VERSION_CODES.ICE_CREAM_SANDWICH_MR1 && Function.openCategory(this, primaryCode)) {
712      Log.info("open category");
713    } else {
714      String s = Event.getCodeText(primaryCode);
715      //Log.info("send Key="+primaryCode+",s="+s);
716      if (!Function.isEmpty(s)) commitText(s);
717      else sendDownUpKeyEvents(primaryCode);
718    }
719  }

721  public void commitEventText(CharSequence text) {
722    String s = text.toString();
723    if (!Event.containsSend(s)) {
724      commitText(s);
725      return;
726    } else {
727      int i = 0, n = s.length(), start = 0, end = 0;
728      start = s.indexOf("{", i);
729      end = s.indexOf("}", start);
730      while (start >= 0 && end > start) {
731        commitText(s.substring(i, start));
732        if (start + 1 == end) commitText("{}");
733        else {
734          int[] sends = Event.parseSend(s.substring(start + 1, end));
735          if (sends[0] + sends[1] > 0) onKey(sends[0], sends[1]);
736        }
737        i = end + 1;
738        start = s.indexOf("{", i);
739        end = s.indexOf("}", start);
740      }
741      if (i < n) commitText(s.substring(i, n));
742    }
743  }

745  public void onText(CharSequence text) { //軟鍵盤
746    Log.info("onText="+text);
747    mEffect.speakKey(text);
748    String s = text.toString();
749    boolean b = s.endsWith("{Left}"); //用於前臺左移光標
750    if (b) text = s.substring(0, s.length() - "{Left}".length());
751    Rime.onText(text);
752    if(!commitText() && !isComposing()) commitEventText(text);
753    if (b) onKey(KeyEvent.KEYCODE_DPAD_LEFT, 0);
754    updateComposing();
755  }

787  public void onPickCandidate(int i) {
788    // Commit the picked candidate and suggest its following words.
789    if (!isComposing()) {
790      if (i >=0) {
791        Rime.toggleOption(i);
792        updateComposing();
793      }
794    } else if (i == -4) onKey(KeyEvent.KEYCODE_PAGE_UP, 0);
795    else if (i == -5) onKey(KeyEvent.KEYCODE_PAGE_DOWN, 0);
796    else if (Rime.selectCandidate(i)) {
797      commitText();
798      updateComposing();
799    }
800  }

802  /** 更新Rime的中西文狀態、編輯區文本 */
803  public void updateComposing() {
804    InputConnection ic = getCurrentInputConnection();
805    if (inlinePreedit != Config.INLINE_NONE) { //嵌入模式
806      String s = null;
807      switch (inlinePreedit) {
808        case Config.INLINE_PREVIEW:
809          s = Rime.getComposingText();
810          break;
811        case Config.INLINE_COMPOSITION:
812          s = Rime.getCompositionText();
813          break;
814        case Config.INLINE_INPUT:
815          s = Rime.RimeGetInput();
816          break;
817      }
818      if (s == null) s = "";
819      if (ic != null) {
820        CharSequence cs = ic.getSelectedText(0);
821        if (cs == null || !Function.isEmpty(s)) {
822          // 無選中文本或編碼不爲空時更新編輯區
823          ic.setComposingText(s, 1);
824        }
825      }
826    }
827    if (ic != null && !isWinFixed() && VERSION.SDK_INT >= VERSION_CODES.LOLLIPOP) cursorUpdated = ic.requestCursorUpdates(1);
828    if (mCandidateContainer != null) {
829      //setCandidatesViewShown(canCompose); //InputType爲0x80000時無候選條
830      int start_num = mComposition.setWindow(min_length);
831      mCandidate.setText(start_num);
832      if (isWinFixed() || !cursorUpdated) mFloatingWindowTimer.postShowFloatingWindow();
833    }
834    if (mKeyboardView != null) mKeyboardView.invalidateComposingKeys();
835  }

863  private boolean handleOption(int keyCode) {
864    if (keyCode == KeyEvent.KEYCODE_MENU) {
865      if (mOptionsDialog != null && mOptionsDialog.isShowing()) return true; //對話框單例
866      AlertDialog.Builder builder = new AlertDialog.Builder(this)
867      .setTitle(R.string.ime_name)
868      .setIcon(R.drawable.icon)
869      .setCancelable(true)
870      .setNegativeButton(R.string.other_ime, new DialogInterface.OnClickListener() {
871        public void onClick(DialogInterface di, int id) {
872          di.dismiss();
873          ((InputMethodManager) getSystemService(INPUT_METHOD_SERVICE)).showInputMethodPicker();
874        }
875      })
876      .setPositiveButton(R.string.set_ime, new DialogInterface.OnClickListener() {
877        public void onClick(DialogInterface di, int id) {
878          Function.showPrefDialog(Trime.this); //全局設置
879          di.dismiss();
880        }
881      });
882      if (Rime.isEmpty()) builder.setMessage(R.string.no_schemas); //提示安裝碼表
883      else {
884        builder.setNeutralButton(R.string.pref_schemas, new DialogInterface.OnClickListener() {
885          public void onClick(DialogInterface di, int id) {
886            showSchemaDialog(); //部署方案
887            di.dismiss();
888          }
889        });
890        builder.setSingleChoiceItems(Rime.getSchemaNames(), Rime.getSchemaIndex(),
891          new DialogInterface.OnClickListener() {
892            public void onClick(DialogInterface di, int id) {
893              di.dismiss();
894              Rime.selectSchema(id); //切換方案
895              updateRimeOption();
896            }
897        });
898      }
899      mOptionsDialog = builder.create();
900      showDialog(mOptionsDialog);
901      return true;
902    }
903    return false;
904  }

908   * 
909   * @param keyCode {@link KeyEvent#getKeyCode() 鍵碼}

915        commitText("\n");

917        sendKeyChar('\n');
918      }
919      return true;
