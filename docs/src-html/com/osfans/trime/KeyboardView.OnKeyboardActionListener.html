072    /** 處理按鍵、觸摸等輸入事件 */
073    public interface OnKeyboardActionListener {
074        
075        /**
076         * Called when the user presses a key. This is sent before the {@link #onKey} is called.
077         * For keys that repeat, this is only called once.
078         * @param primaryCode the unicode of the key being pressed. If the touch is not on a valid
079         * key, the value will be zero.
080         */
081        void onPress(int primaryCode);
082        
083        /**
084         * Called when the user releases a key. This is sent after the {@link #onKey} is called.
085         * For keys that repeat, this is only called once.
086         * @param primaryCode the code of the key that was released
087         */
088        void onRelease(int primaryCode);
089
090        void onEvent(Event event);
091
092        /**
093         * Send a key press to the listener.
094         * @param primaryCode this is the key that was pressed
095         * @param mask the codes for all the possible alternative keys
096         * with the primary code being the first. If the primary key code is
097         * a single character such as an alphabet or number or symbol, the alternatives
098         * will include other characters that may be on the same key or adjacent keys.
099         * These codes are useful to correct for accidental presses of a key adjacent to
100         * the intended key.
101         */
102        void onKey(int primaryCode, int mask);
103
104        /**
105         * Sends a sequence of characters to the listener.
106         * @param text the sequence of characters to be displayed.
107         */
108        void onText(CharSequence text);
109        
110        /**
111         * Called when the user quickly moves the finger from right to left.
112         */
113        void swipeLeft();
114        
115        /**
116         * Called when the user quickly moves the finger from left to right.
117         */
118        void swipeRight();
119        
120        /**
121         * Called when the user quickly moves the finger from up to down.
122         */
123        void swipeDown();
124        
125        /**
126         * Called when the user quickly moves the finger from down to up.
127         */
128        void swipeUp();
129    }

260    Handler mHandler = new Handler() {
261        @Override
262        public void handleMessage(Message msg) {
263            switch (msg.what) {
264                case MSG_SHOW_PREVIEW:
265                    showKey(msg.arg1, msg.arg2);
266                    break;
267                case MSG_REMOVE_PREVIEW:
268                    mPreviewText.setVisibility(INVISIBLE);
269                    break;
270                case MSG_REPEAT:
271                    if (repeatKey()) {
272                        Message repeat = Message.obtain(this, MSG_REPEAT);
273                        sendMessageDelayed(repeat, REPEAT_INTERVAL);
274                    }
275                    break;
276                case MSG_LONGPRESS:
277                    openPopupIfRequired((MotionEvent) msg.obj);
278                    break;
279            }
280        }
281    };

396    private void initGestureDetector() {
397        mGestureDetector = new GestureDetector(getContext(), new GestureDetector.SimpleOnGestureListener() {
398            @Override
399            public boolean onFling(MotionEvent me1, MotionEvent me2, 
400                    float velocityX, float velocityY) {
401                if (mPossiblePoly) return false;
402                final float absX = Math.abs(velocityX);
403                final float absY = Math.abs(velocityY);
404                float deltaX = me2.getX() - me1.getX();
405                float deltaY = me2.getY() - me1.getY();
406                int travelX = 0; //getWidth() / 2; // Half the keyboard width
407                int travelY = 0; //getHeight() / 2; // Half the keyboard height
408                mSwipeTracker.computeCurrentVelocity(10);
409                final float endingVelocityX = mSwipeTracker.getXVelocity();
410                final float endingVelocityY = mSwipeTracker.getYVelocity();
411                boolean sendDownKey = false;
412                int type = 0;
413                if (velocityX > mSwipeThreshold && absY < absX && deltaX > travelX) {
414                    if (mDisambiguateSwipe && endingVelocityX < velocityX / 4) {
415                        sendDownKey = true;
416                        type = Key.SWIPE_RIGHT;
417                    } else {
418                        swipeRight();
419                        return true;
420                    }
421                } else if (velocityX < -mSwipeThreshold && absY < absX && deltaX < -travelX) {
422                    if (mDisambiguateSwipe && endingVelocityX > velocityX / 4) {
423                        sendDownKey = true;
424                        type = Key.SWIPE_LEFT;
425                    } else {
426                        swipeLeft();
427                        return true;
428                    }
429                } else if (velocityY < -mSwipeThreshold && absX < absY && deltaY < -travelY) {
430                    if (mDisambiguateSwipe && endingVelocityY > velocityY / 4) {
431                        sendDownKey = true;
432                        type = Key.SWIPE_UP;
433                    } else {
434                        swipeUp();
435                        return true;
436                    }
437                } else if (velocityY > mSwipeThreshold && absX < absY / 2 && deltaY > travelY) {
438                    if (mDisambiguateSwipe && endingVelocityY < velocityY / 4) {
439                        sendDownKey = true;
440                        type = Key.SWIPE_DOWN;
441                    } else {
442                        swipeDown();
443                        return true;
444                    }
445                }
446
447                if (sendDownKey) {
448                    showPreview(NOT_A_KEY);
449                    showPreview(mDownKey, type);
450                    detectAndSendKey(mDownKey, mStartX, mStartY, me1.getEventTime(), type);
451                    return true;
452                }
453                return false;
454            }
455        });
456
457        mGestureDetector.setIsLongpressEnabled(false);
458    }

472    /**
473     * Attaches a keyboard to this view. The keyboard can be switched at any time and the
474     * view will re-layout itself to accommodate the keyboard.
475     * @see Keyboard
476     * @see #getKeyboard()
477     * @param keyboard the keyboard to display in this view
478     */
479    public void setKeyboard(Keyboard keyboard) {
480        if (mKeyboard != null) {
481            showPreview(NOT_A_KEY);
482        }
483        // Remove any pending messages
484        removeMessages();
485        mRepeatKeyIndex = NOT_A_KEY;
486        mKeyboard = keyboard;
487        List<Key> keys = mKeyboard.getKeys();
488        mKeys = keys.toArray(new Key[keys.size()]);
489        requestLayout();
490        // Hint to reallocate the buffer if the size changed
491        mKeyboardChanged = true;
492        invalidateAllKeys();
493        computeProximityThreshold(keyboard);
494        mMiniKeyboardCache.clear(); // Not really necessary to do every time, but will free up views
495        // Switching to a different keyboard should abort any pending keys so that the key up
496        // doesn't get delivered to the old or new keyboard
497        mAbortKey = true; // Until the next ACTION_DOWN
498    }

634    /**
635     * 計算水平和豎直方向的相鄰按鍵中心的平均距離的平方，這樣不需要做開方運算
636     * @param keyboard 鍵盤
637     */
638    private void computeProximityThreshold(Keyboard keyboard) {
639        if (keyboard == null) return;
640        final Key[] keys = mKeys;
641        if (keys == null) return;
642        int length = keys.length;
643        int dimensionSum = 0;
644        for (int i = 0; i < length; i++) {
645            Key key = keys[i];
646            dimensionSum += Math.min(key.width, key.height) + key.gap;
647        }
648        if (dimensionSum < 0 || length == 0) return;
649        mProximityThreshold = (int) (dimensionSum * Keyboard.SEARCH_DISTANCE / length);
650        mProximityThreshold *= mProximityThreshold; // Square it
651    }

672    private void onBufferDraw() {
673        if (mBuffer == null || mKeyboardChanged) {
674            if (mBuffer == null || mKeyboardChanged &&
675                    (mBuffer.getWidth() != getWidth() || mBuffer.getHeight() != getHeight())) {
676                // Make sure our bitmap is at least 1x1
677                final int width = Math.max(1, getWidth());
678                final int height = Math.max(1, getHeight());
679                mBuffer = Bitmap.createBitmap(width, height, Bitmap.Config.ARGB_8888);
680                mCanvas = new Canvas(mBuffer);
681            }
682            invalidateAllKeys();
683            mKeyboardChanged = false;
684        }
685        final Canvas canvas = mCanvas;
686        canvas.clipRect(mDirtyRect, Op.REPLACE);
687        
688        if (mKeyboard == null) return;
689        
690        final Paint paint = mPaint;
691        Drawable keyBackground;
692        final Rect clipRegion = mClipRegion;
693        final Rect padding = mPadding;
694        final int kbdPaddingLeft = getPaddingLeft();
695        final int kbdPaddingTop = getPaddingTop();
696        final Key[] keys = mKeys;
697        final Key invalidKey = mInvalidatedKey;
698
699        boolean drawSingleKey = false;
700        if (invalidKey != null && canvas.getClipBounds(clipRegion)) {
701          // Is clipRegion completely contained within the invalidated key?
702          if (invalidKey.x + kbdPaddingLeft - 1 <= clipRegion.left &&
703                  invalidKey.y + kbdPaddingTop - 1 <= clipRegion.top &&
704                  invalidKey.x + invalidKey.width + kbdPaddingLeft + 1 >= clipRegion.right &&
705                  invalidKey.y + invalidKey.height + kbdPaddingTop + 1 >= clipRegion.bottom) {
706              drawSingleKey = true;
707          }
708        }
709        canvas.drawColor(0x00000000, PorterDuff.Mode.CLEAR);
710        final int keyCount = keys.length;
711        final float symbolBase = padding.top - mPaintSymbol.getFontMetrics().top;
712        final float hintBase =  - padding.bottom - mPaintSymbol.getFontMetrics().bottom;
713        for (int i = 0; i < keyCount; i++) {
714            final Key key = keys[i];
715            if (drawSingleKey && invalidKey != key) {
716                continue;
717            }
718            int[] drawableState = key.getCurrentDrawableState();
719            keyBackground = key.getBackColorForState(drawableState);
720            if (keyBackground == null) {
721                try {
722                    int index = (int) getStateDrawableIndex.invoke(mKeyBackColor, drawableState);
723                    keyBackground = (Drawable) getStateDrawable.invoke(mKeyBackColor, index);
724                } catch (Exception ex){
725                }
726            }
727            if (keyBackground instanceof GradientDrawable) {
728                ((GradientDrawable)keyBackground).setCornerRadius(key.round_corner != null ? key.round_corner : mRoundCorner);
729            }
730            Integer color = key.getTextColorForState(drawableState);
731            mPaint.setColor(color != null ? color : mKeyTextColor.getColorForState(drawableState, 0));
732            color = key.getSymbolColorForState(drawableState);
733            mPaintSymbol.setColor(color != null ? color : (key.pressed ? hilited_key_symbol_color: key_symbol_color));
734
735            // Switch the character to uppercase if shift is pressed
736            String label = key.getLabel();
737            String hint = key.hint;
738            int left = (key.width - padding.left - padding.right) / 2 + padding.left;
739            int top = padding.top;
740            
741            final Rect bounds = keyBackground.getBounds();
742            if (key.width != bounds.right || 
743                    key.height != bounds.bottom) {
744                keyBackground.setBounds(0, 0, key.width, key.height);
745            }
746            canvas.translate(key.x + kbdPaddingLeft, key.y + kbdPaddingTop);
747            keyBackground.draw(canvas);
748            
749            if (!Function.isEmpty(label)) {
750                // For characters, use large font. For labels like "Done", use small font.
751                if (key.key_text_size != null) {
752                  paint.setTextSize(key.key_text_size);
753                } else {
754                  paint.setTextSize(label.length() > 1 ? mLabelTextSize: mKeyTextSize);
755                }
756                // Draw a drop shadow for the text
757                paint.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
758                // Draw the text
759                canvas.drawText(label, left,
760                    (key.height - padding.top - padding.bottom) / 2
761                            + (paint.getTextSize() - paint.descent()) / 2 + top,
762                    paint);
763
764                if (key.getLongClick() != null) {
765                    mPaintSymbol.setTextSize(key.symbol_text_size != null ? key.symbol_text_size : mSymbolSize);
766                    mPaintSymbol.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
767                    canvas.drawText(key.getSymbolLabel(), left, symbolBase, mPaintSymbol);
768                }
769
770                if (!Function.isEmpty(hint)) {
771                    mPaintSymbol.setShadowLayer(mShadowRadius, 0, 0, mShadowColor);
772                    canvas.drawText(hint, left, key.height + hintBase, mPaintSymbol);
773                }
774
775                // Turn off drop shadow
776                paint.setShadowLayer(0, 0, 0, 0);
777            }
778            canvas.translate(-key.x - kbdPaddingLeft, -key.y - kbdPaddingTop);
779        }
780        mInvalidatedKey = null;
781        // Overlay a dark rectangle to dim the keyboard
782        if (mMiniKeyboardOnScreen) {
783            paint.setColor((int) (mBackgroundDimAmount * 0xFF) << 24);
784            canvas.drawRect(0, 0, getWidth(), getHeight(), paint);
785        }
786
787        if (DEBUG && mShowTouchPoints) {
788            paint.setAlpha(128);
789            paint.setColor(0xFFFF0000);
790            canvas.drawCircle(mStartX, mStartY, 3, paint);
791            canvas.drawLine(mStartX, mStartY, mLastX, mLastY, paint);
792            paint.setColor(0xFF0000FF);
793            canvas.drawCircle(mLastX, mLastY, 3, paint);
794            paint.setColor(0xFF00FF00);
795            canvas.drawCircle((mStartX + mLastX) / 2, (mStartY + mLastY) / 2, 2, paint);
796        }
797        
798        mDrawPending = false;
799        mDirtyRect.setEmpty();
800    }

802    private int getKeyIndices(int x, int y, int[] allKeys) {
803        final Key[] keys = mKeys;
804        int primaryIndex = NOT_A_KEY;
805        int closestKey = NOT_A_KEY;
806        int closestKeyDist = mProximityThreshold + 1;
807        java.util.Arrays.fill(mDistances, Integer.MAX_VALUE);
808        int [] nearestKeyIndices = mKeyboard.getNearestKeys(x, y);
809        final int keyCount = nearestKeyIndices.length;
810        for (int i = 0; i < keyCount; i++) {
811            final Key key = keys[nearestKeyIndices[i]];
812            int dist = 0;
813            boolean isInside = key.isInside(x,y);
814            if (isInside) {
815                primaryIndex = nearestKeyIndices[i];
816            }
817
818            if ((mProximityCorrectOn 
819                    && (dist = key.squaredDistanceFrom(x, y)) < mProximityThreshold) 
820                    || isInside) {
821                // Find insertion point
822                final int nCodes = 1;
823                if (dist < closestKeyDist) {
824                    closestKeyDist = dist;
825                    closestKey = nearestKeyIndices[i];
826                }
827                
828                if (allKeys == null) continue;
829                
830                for (int j = 0; j < mDistances.length; j++) {
831                    if (mDistances[j] > dist) {
832                        // Make space for nCodes codes
833                        System.arraycopy(mDistances, j, mDistances, j + nCodes,
834                                mDistances.length - j - nCodes);
835                        System.arraycopy(allKeys, j, allKeys, j + nCodes,
836                                allKeys.length - j - nCodes);
837                        allKeys[j] = key.getCode();
838                        mDistances[j] = dist;
839                        break;
840                    }
841                }
842            }
843        }
844        if (primaryIndex == NOT_A_KEY) {
845            primaryIndex = closestKey;
846        }
847        return primaryIndex;
848    }

852            if (mComboCount > 9 ) mComboCount = 9;
853            mComboCodes[mComboCount++] = code;

858                    mKeyboardActionListener.onRelease(mComboCodes[i]);
859                }
860                mComboCount = 0;
861            }

889    private void showPreview(int keyIndex, int type) {
890        int oldKeyIndex = mCurrentKeyIndex;
891        final PopupWindow previewPopup = mPreviewPopup;
892        
893        mCurrentKeyIndex = keyIndex;
894        // Release the old key and press the new key
895        final Key[] keys = mKeys;
896        if (oldKeyIndex != mCurrentKeyIndex) {
897            if (oldKeyIndex != NOT_A_KEY && keys.length > oldKeyIndex) {
898                Key oldKey = keys[oldKeyIndex];
899                oldKey.onReleased(mCurrentKeyIndex == NOT_A_KEY);
900                invalidateKey(oldKeyIndex);
901            }
902            if (mCurrentKeyIndex != NOT_A_KEY && keys.length > mCurrentKeyIndex) {
903                Key newKey = keys[mCurrentKeyIndex];
904                newKey.onPressed();
905                invalidateKey(mCurrentKeyIndex);
906            }
907        }
908        // If key changed and preview is on ...
909        if (oldKeyIndex != mCurrentKeyIndex && mShowPreview) {
910            mHandler.removeMessages(MSG_SHOW_PREVIEW);
911            if (previewPopup.isShowing()) {
912                if (keyIndex == NOT_A_KEY) {
913                    mHandler.sendMessageDelayed(mHandler
914                            .obtainMessage(MSG_REMOVE_PREVIEW),
915                            DELAY_AFTER_PREVIEW);
916                }
917            }
918            if (keyIndex != NOT_A_KEY) {
919                if (previewPopup.isShowing() && mPreviewText.getVisibility() == VISIBLE) {
920                    // Show right away, if it's already visible and finger is moving around
921                    showKey(keyIndex, type);
922                } else {
923                    mHandler.sendMessageDelayed(
924                            mHandler.obtainMessage(MSG_SHOW_PREVIEW, keyIndex, type),
925                            DELAY_BEFORE_PREVIEW);
926                }
927            }
928        }
929    }

936    private void showKey(final int keyIndex, int type) {
937        final PopupWindow previewPopup = mPreviewPopup;
938        final Key[] keys = mKeys;
939        if (keyIndex < 0 || keyIndex >= mKeys.length) return;
940        Key key = keys[keyIndex];
941        mPreviewText.setCompoundDrawables(null, null, null, null);
942        mPreviewText.setText(key.getPreviewText(type));
943        mPreviewText.measure(MeasureSpec.makeMeasureSpec(0, MeasureSpec.UNSPECIFIED), 
944                MeasureSpec.makeMeasureSpec(0, MeasureSpec.UNSPECIFIED));
945        int popupWidth = Math.max(mPreviewText.getMeasuredWidth(), key.width 
946                + mPreviewText.getPaddingLeft() + mPreviewText.getPaddingRight());
947        final int popupHeight = mPreviewHeight;
948        LayoutParams lp = mPreviewText.getLayoutParams();
949        if (lp != null) {
950            lp.width = popupWidth;
951            lp.height = popupHeight;
952        }
953        if (!mPreviewCentered) {
954            mPopupPreviewX = key.x - mPreviewText.getPaddingLeft() + getPaddingLeft();
955            mPopupPreviewY = key.y - popupHeight + mPreviewOffset;
956        } else {
957            // TODO: Fix this if centering is brought back
958            mPopupPreviewX = 160 - mPreviewText.getMeasuredWidth() / 2;
959            mPopupPreviewY = - mPreviewText.getMeasuredHeight();
960        }
961        mHandler.removeMessages(MSG_REMOVE_PREVIEW);
962        getLocationInWindow(mCoordinates);
963        mCoordinates[0] += mMiniKeyboardOffsetX; // Offset may be zero
964        mCoordinates[1] += mMiniKeyboardOffsetY; // Offset may be zero
965
966        // Set the preview background state
967        mPreviewText.getBackground().setState(
968                key.popupResId != 0 ? LONG_PRESSABLE_STATE_SET : EMPTY_STATE_SET);
969        mPopupPreviewX += mCoordinates[0];
970        mPopupPreviewY += mCoordinates[1];
971
972        // If the popup cannot be shown above the key, put it on the side
973        getLocationOnScreen(mCoordinates);
974        if (mPopupPreviewY + mCoordinates[1] < 0) {
975            // If the key you're pressing is on the left side of the keyboard, show the popup on
976            // the right, offset by enough to see at least one key to the left/right.
977            if (key.x + key.width <= getWidth() / 2) {
978                mPopupPreviewX += (int) (key.width * 2.5);
979            } else {
980                mPopupPreviewX -= (int) (key.width * 2.5);
981            }
982            mPopupPreviewY += popupHeight;
983        }
984
985        if (previewPopup.isShowing()) {
986            previewPopup.update(mPopupPreviewX, mPopupPreviewY,
987                    popupWidth, popupHeight);
988        } else {
989            previewPopup.setWidth(popupWidth);
990            previewPopup.setHeight(popupHeight);
991            previewPopup.showAtLocation(mPopupParent, Gravity.NO_GRAVITY, 
992                    mPopupPreviewX, mPopupPreviewY);
993        }
994        mPreviewText.setVisibility(VISIBLE);
995    }

1149            if (key.getClick().repeatable) return false;
1150            if (key.getLongClick() != null) {
1151              mKeyboardActionListener.onEvent(key.getLongClick());
1152              return true;
1153            }
1154            if (key.isShift()) {
1155              setShifted(!key.on, !key.on);
1156              return true;
1157            }
1158        }
1159        return false;
1160    }

1189    @Override
1190    public boolean onTouchEvent(MotionEvent me) {
1191        // Convert multi-pointer up/down events to single up/down events to 
1192        // deal with the typical multi-pointer behavior of two-thumb typing
1193        final int index = me.getActionIndex();
1194        final int pointerCount = me.getPointerCount();
1195        final int action = me.getActionMasked();
1196        boolean result = false;
1197        final long now = me.getEventTime();
1198
1199        mComboMode = false;
1200        if (action == MotionEvent.ACTION_DOWN || action == MotionEvent.ACTION_CANCEL) {
1201            mComboCount = 0;
1202        } else if (pointerCount > 1 || action == MotionEvent.ACTION_POINTER_DOWN || action == MotionEvent.ACTION_POINTER_UP) {
1203            mComboMode = true;
1204        }
1205
1206        if (action == MotionEvent.ACTION_POINTER_UP || (mOldPointerCount > 1 && action == MotionEvent.ACTION_UP)) {
1207            //並擊鬆開前的虛擬按鍵事件
1208            MotionEvent ev = MotionEvent.obtain(now, now, MotionEvent.ACTION_POINTER_DOWN,
1209                me.getX(index), me.getY(index), me.getMetaState());
1210            result = onModifiedTouchEvent(ev, false);
1211            ev.recycle();
1212        }
1213
1214        if (action == MotionEvent.ACTION_POINTER_DOWN) {
1215            //並擊中的按鍵事件，需要按鍵提示
1216            MotionEvent ev = MotionEvent.obtain(now, now,
1217                MotionEvent.ACTION_DOWN,
1218                me.getX(index), me.getY(index), me.getMetaState());
1219            result = onModifiedTouchEvent(ev, false);
1220            ev.recycle();
1221        } else {
1222            result = onModifiedTouchEvent(me, false);
1223        }
1224
1225        if (action != MotionEvent.ACTION_MOVE) mOldPointerCount = pointerCount;
1226        performClick();
1227        return result;
1228    }

1230    private boolean onModifiedTouchEvent(MotionEvent me, boolean possiblePoly) {
1231        final int pointerCount = me.getPointerCount();
1232        final int index = me.getActionIndex();
1233        int touchX = (int) me.getX(index) - getPaddingLeft();
1234        int touchY = (int) me.getY(index) - getPaddingTop();
1235        if (touchY >= -mVerticalCorrection)
1236            touchY += mVerticalCorrection;
1237        final int action = me.getActionMasked();
1238        final long eventTime = me.getEventTime();
1239        int keyIndex = getKeyIndices(touchX, touchY, null);
1240        mPossiblePoly = possiblePoly;
1241
1242        // Track the last few movements to look for spurious swipes.
1243        if (action == MotionEvent.ACTION_DOWN) mSwipeTracker.clear();
1244        mSwipeTracker.addMovement(me);
1245
1246        // Ignore all motion events until a DOWN.
1247        if (mAbortKey
1248                && action != MotionEvent.ACTION_DOWN && action != MotionEvent.ACTION_CANCEL) {
1249            return true;
1250        }
1251
1252        if (mGestureDetector.onTouchEvent(me)) {
1253            showPreview(NOT_A_KEY);
1254            mHandler.removeMessages(MSG_REPEAT);
1255            mHandler.removeMessages(MSG_LONGPRESS);
1256            return true;
1257        }
1258        
1259        // Needs to be called after the gesture detector gets a turn, as it may have
1260        // displayed the mini keyboard
1261        if (mMiniKeyboardOnScreen && action != MotionEvent.ACTION_CANCEL) {
1262            return true;
1263        }
1264
1265        switch (action) {
1266            case MotionEvent.ACTION_DOWN:
1267            case MotionEvent.ACTION_POINTER_DOWN:
1268                mAbortKey = false;
1269                mStartX = touchX;
1270                mStartY = touchY;
1271                mLastCodeX = touchX;
1272                mLastCodeY = touchY;
1273                mLastKeyTime = 0;
1274                mCurrentKeyTime = 0;
1275                mLastKey = NOT_A_KEY;
1276                mCurrentKey = keyIndex;
1277                mDownKey = keyIndex;
1278                mDownTime = me.getEventTime();
1279                mLastMoveTime = mDownTime;
1280                if (action == MotionEvent.ACTION_POINTER_DOWN) break; //並擊鬆開前的虛擬按鍵事件
1281                checkMultiTap(eventTime, keyIndex);
1282                mKeyboardActionListener.onPress(keyIndex != NOT_A_KEY ? 
1283                        mKeys[keyIndex].getCode() : 0);
1284                if (mCurrentKey >= 0 && mKeys[mCurrentKey].getClick().repeatable) {
1285                    mRepeatKeyIndex = mCurrentKey;
1286                    Message msg = mHandler.obtainMessage(MSG_REPEAT);
1287                    mHandler.sendMessageDelayed(msg, REPEAT_START_DELAY);
1288                    repeatKey();
1289                    // Delivering the key could have caused an abort
1290                    if (mAbortKey) {
1291                        mRepeatKeyIndex = NOT_A_KEY;
1292                        break;
1293                    }
1294                }
1295                if (mCurrentKey != NOT_A_KEY) {
1296                    Message msg = mHandler.obtainMessage(MSG_LONGPRESS, me);
1297                    mHandler.sendMessageDelayed(msg, LONGPRESS_TIMEOUT);
1298                }
1299                showPreview(keyIndex, 0);
1300                break;
1301
1302            case MotionEvent.ACTION_MOVE:
1303                boolean continueLongPress = false;
1304                if (keyIndex != NOT_A_KEY) {
1305                    if (mCurrentKey == NOT_A_KEY) {
1306                        mCurrentKey = keyIndex;
1307                        mCurrentKeyTime = eventTime - mDownTime;
1308                    } else {
1309                        if (keyIndex == mCurrentKey) {
1310                            mCurrentKeyTime += eventTime - mLastMoveTime;
1311                            continueLongPress = true;
1312                        } else if (mRepeatKeyIndex == NOT_A_KEY) {
1313                            resetMultiTap();
1314                            mLastKey = mCurrentKey;
1315                            mLastCodeX = mLastX;
1316                            mLastCodeY = mLastY;
1317                            mLastKeyTime =
1318                                    mCurrentKeyTime + eventTime - mLastMoveTime;
1319                            mCurrentKey = keyIndex;
1320                            mCurrentKeyTime = 0;
1321                        }
1322                    }
1323                }
1324                if (!mComboMode && !continueLongPress) {
1325                    // Cancel old longpress
1326                    mHandler.removeMessages(MSG_LONGPRESS);
1327                    // Start new longpress if key has changed
1328                    if (keyIndex != NOT_A_KEY) {
1329                        Message msg = mHandler.obtainMessage(MSG_LONGPRESS, me);
1330                        mHandler.sendMessageDelayed(msg, LONGPRESS_TIMEOUT);
1331                    }
1332                }
1333                showPreview(mCurrentKey);
1334                mLastMoveTime = eventTime;
1335                break;
1336
1337            case MotionEvent.ACTION_UP:
1338            case MotionEvent.ACTION_POINTER_UP:
1339                removeMessages();
1340                if (keyIndex == mCurrentKey) {
1341                    mCurrentKeyTime += eventTime - mLastMoveTime;
1342                } else {
1343                    resetMultiTap();
1344                    mLastKey = mCurrentKey;
1345                    mLastKeyTime = mCurrentKeyTime + eventTime - mLastMoveTime;
1346                    mCurrentKey = keyIndex;
1347                    mCurrentKeyTime = 0;
1348                }
1349                if (mCurrentKeyTime < mLastKeyTime && mCurrentKeyTime < DEBOUNCE_TIME
1350                        && mLastKey != NOT_A_KEY) {
1351                    mCurrentKey = mLastKey;
1352                    touchX = mLastCodeX;
1353                    touchY = mLastCodeY;
1354                }
1355                showPreview(NOT_A_KEY);
1356                Arrays.fill(mKeyIndices, NOT_A_KEY);
1357                // If we're not on a repeating key (which sends on a DOWN event)
1358                if (mRepeatKeyIndex == NOT_A_KEY && !mMiniKeyboardOnScreen && !mAbortKey) {
1359                    detectAndSendKey(mCurrentKey, touchX, touchY, eventTime,
1360                        (mOldPointerCount > 1 || mComboMode) ? Key.COMBO : 0);
1361                }
1362                invalidateKey(keyIndex);
1363                mRepeatKeyIndex = NOT_A_KEY;
1364                break;
1365            case MotionEvent.ACTION_CANCEL:
1366                removeMessages();
1367                dismissPopupKeyboard();
1368                mAbortKey = true;
1369                showPreview(NOT_A_KEY);
1370                invalidateKey(mCurrentKey);
1371                break;
1372        }
1373        mLastX = touchX;
1374        mLastY = touchY;
1375        return true;
1376    }

1455    /** 識別滑動手勢 */
1456    private static class SwipeTracker {
1457
1458        static final int NUM_PAST = 4;
1459        static final int LONGEST_PAST_TIME = 200;
1460
1461        final float mPastX[] = new float[NUM_PAST];
1462        final float mPastY[] = new float[NUM_PAST];
1463        final long mPastTime[] = new long[NUM_PAST];
1464
1465        float mYVelocity;
1466        float mXVelocity;
1467
1468        public void clear() {
1469            mPastTime[0] = 0;
1470        }
1471
1472        public void addMovement(MotionEvent ev) {
1473            long time = ev.getEventTime();
1474            final int N = ev.getHistorySize();
1475            for (int i=0; i<N; i++) {
1476                addPoint(ev.getHistoricalX(i), ev.getHistoricalY(i),
1477                        ev.getHistoricalEventTime(i));
1478            }
1479            addPoint(ev.getX(), ev.getY(), time);
1480        }
1481
1482        private void addPoint(float x, float y, long time) {
1483            int drop = -1;
1484            int i;
1485            final long[] pastTime = mPastTime;
1486            for (i=0; i<NUM_PAST; i++) {
1487                if (pastTime[i] == 0) {
1488                    break;
1489                } else if (pastTime[i] < time-LONGEST_PAST_TIME) {
1490                    drop = i;
1491                }
1492            }
1493            if (i == NUM_PAST && drop < 0) {
1494                drop = 0;
1495            }
1496            if (drop == i) drop--;
1497            final float[] pastX = mPastX;
1498            final float[] pastY = mPastY;
1499            if (drop >= 0) {
1500                final int start = drop+1;
1501                final int count = NUM_PAST-drop-1;
1502                System.arraycopy(pastX, start, pastX, 0, count);
1503                System.arraycopy(pastY, start, pastY, 0, count);
1504                System.arraycopy(pastTime, start, pastTime, 0, count);
1505                i -= (drop+1);
1506            }
1507            pastX[i] = x;
1508            pastY[i] = y;
1509            pastTime[i] = time;
1510            i++;
1511            if (i < NUM_PAST) {
1512                pastTime[i] = 0;
1513            }
1514        }
1515
1516        public void computeCurrentVelocity(int units) {
1517            computeCurrentVelocity(units, Float.MAX_VALUE);
1518        }
1519
1520        public void computeCurrentVelocity(int units, float maxVelocity) {
1521            final float[] pastX = mPastX;
1522            final float[] pastY = mPastY;
1523            final long[] pastTime = mPastTime;
1524
1525            final float oldestX = pastX[0];
1526            final float oldestY = pastY[0];
1527            final long oldestTime = pastTime[0];
1528            float accumX = 0;
1529            float accumY = 0;
1530            int N=0;
1531            while (N < NUM_PAST) {
1532                if (pastTime[N] == 0) {
1533                    break;
1534                }
1535                N++;
1536            }
1537
1538            for (int i=1; i < N; i++) {
1539                final int dur = (int)(pastTime[i] - oldestTime);
1540                if (dur == 0) continue;
1541                float dist = pastX[i] - oldestX;
1542                float vel = (dist/dur) * units;   // pixels/frame.
1543                if (accumX == 0) accumX = vel;
1544                else accumX = (accumX + vel) * .5f;
1545
1546                dist = pastY[i] - oldestY;
1547                vel = (dist/dur) * units;   // pixels/frame.
1548                if (accumY == 0) accumY = vel;
1549                else accumY = (accumY + vel) * .5f;
1550            }
1551            mXVelocity = accumX < 0.0f ? Math.max(accumX, -maxVelocity)
1552                    : Math.min(accumX, maxVelocity);
1553            mYVelocity = accumY < 0.0f ? Math.max(accumY, -maxVelocity)
1554                    : Math.min(accumY, maxVelocity);
1555        }
1556
1557        public float getXVelocity() {
1558            return mXVelocity;
1559        }
1560
1561        public float getYVelocity() {
1562            return mYVelocity;
1563        }
1564    }
