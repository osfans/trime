017package com.osfans.trime;
018
019import android.app.AlertDialog;
020import android.content.DialogInterface;
021import android.content.Context;
022import android.util.Log;
023import android.widget.Toast;
024
025import java.util.Arrays;
026
027/** 顯示輸入法內置數據列表，並回廠選中的數據 */
028public class ResetDialog {
029  /** 內置數據列表 */
030  String[] items;
031  /** 列表勾選狀態 */
032  boolean[] checked;
033  /** 回廠對話框 */
034  AlertDialog dialog;
035  Context context;
036
037  /** 回廠選中的數據 */
038  private void select() {
039    if (items == null) return;
040    boolean ret = true;
041    int n = items.length;
042    for (int i = 0; i < n; i++) {
043      if (checked[i]) {
044        ret = Config.copyFileOrDir(context, "rime/" + items[i], true);
045      }
046    }
047    Toast.makeText(context, ret ? R.string.reset_success : R.string.reset_failure, Toast.LENGTH_SHORT).show();
048  }
049
050  public ResetDialog(Context context) {
051    this.context = context;
052    items = Config.list(context, "rime");
053    if (items == null) return;
054    checked = new boolean[items.length];
055    dialog = new AlertDialog.Builder(context)
056      .setTitle(R.string.pref_reset)
057      .setCancelable(true)
058      .setNegativeButton(android.R.string.cancel, null)
059      .setPositiveButton(android.R.string.ok, new DialogInterface.OnClickListener() {
060        public void onClick(DialogInterface di, int id) {
061          select();
062        }
063      })
064      .setMultiChoiceItems(items, checked, new DialogInterface.OnMultiChoiceClickListener() {
065        public void onClick(DialogInterface di, int id, boolean isChecked) {
066          checked[id] = isChecked;
067        }
068      })
069      .create();
070  }
071
072  /**
073   * 獲得回廠對話框
074   * @return 回廠對話框對象
075   */
076  public AlertDialog getDialog() {
077    return dialog;
078  }
079
080  /** 彈出對話框 */
081  public void show() {
082    if (dialog != null) dialog.show();
083  }
084}
