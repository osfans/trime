119  /** Rime方案 */
120  public static class RimeSchema {
121    private String kRightArrow = "→ ";
122    private String kRadioSelected = " ✓";
123
124    Map<String, Object> schema = new HashMap<String, Object>();
125    List<Map<String, Object>> switches = new ArrayList<Map<String, Object>>();
126
127    public RimeSchema(String schema_id) {
128      Object o;
129      o = schema_get_value(schema_id, "schema");
130      if (o == null || !(o instanceof Map)) return;
131      schema = (Map<String, Object>)o;
132      o = schema_get_value(schema_id, "switches");
133      if (o == null || !(o instanceof List)) return;
134      switches = (List<Map<String, Object>>)o;
135      check(); //檢查不在選單中顯示的選項
136    }
137
138    public void check() {
139      if (switches.isEmpty()) return;
140      for (Iterator it = switches.iterator(); it.hasNext();) {
141        Map<String, Object> o = (Map<String, Object>) it.next();
142        if (!o.containsKey("states")) it.remove();
143      }
144    }
145
146    public RimeCandidate[] getCandidates() {
147      if (switches.isEmpty()) return null;
148      RimeCandidate[] candidates = new RimeCandidate[switches.size()];
149      int i = 0;
150      for (Map<String, Object> o: switches) {
151        candidates[i] = new RimeCandidate();
152        List states = (List)o.get("states");
153        Integer value = (Integer)o.get("value");
154        if (value == null) value = 0;
155        candidates[i].text = states.get(value).toString();
156        candidates[i].comment = o.containsKey("options") ? "" : kRightArrow + states.get(1 - value).toString();
157        i++;
158      }
159      return candidates;
160    }
161
162    public void getValue(int session_id) {
163      if (switches.isEmpty()) return; //無方案
164      for (int j = 0; j < switches.size(); j++) {
165        Map<String, Object> o =  switches.get(j);
166        if (o.containsKey("options")) {
167          List<String> options = (List<String>)o.get("options");
168          for (int i = 0; i < options.size(); i++) {
169            String s = options.get(i);
170            if (Rime.get_option(session_id, s)) {
171              o.put("value", i);
172              break;
173            }
174          }
175        } else {
176          o.put("value", Rime.get_option(session_id, o.get("name").toString()) ? 1 : 0);
177        }
178        switches.set(j, o);
179      }
180    }
181
182    public void toggleOption(int i) {
183      if (switches.isEmpty()) return;
184      Map<String, Object> o =  switches.get(i);
185      Integer value = (Integer)o.get("value");
186      if (value == null) value = 0;
187      if (o.containsKey("options")) {
188        List<String> options = (List<String>)o.get("options");
189        Rime.setOption(options.get(value), false);
190        value = (value + 1) % options.size();
191        Rime.setOption(options.get(value), true);
192      } else {
193        value = 1 - value;
194        Rime.setOption(o.get("name").toString(), value == 1);
195      }
196      o.put("value", value);
197      switches.set(i, o);
198    }
199  };

340  public static String[] getSelectLabels() {
341    if (mContext != null && mContext.size() > 0) {
342      if (mContext.select_labels != null) return mContext.select_labels;
343      if (mContext.menu.select_keys != null) return mContext.menu.select_keys.split("\\B");
344      int n = mContext.size();
345      String[] labels = new String[n];
346      for (int i = 0; i < n; i++) {
347        labels[i] = String.valueOf((i + 1)% 10);
348      }
349      return labels;
350    }
351    return null;
352  }

444  public static boolean selectSchema(int id) {
445    int n = mSchemaList.size();
446    if (id < 0 || id >= n) return false;
447    String schema_id = getSchemaId();
448    Map<String, String> m = (Map<String, String>)mSchemaList.get(id);
449    String target = m.get("schema_id");
450    if (target.contentEquals(schema_id)) return false;
451    return selectSchema(target);
452  }

480  public static void onMessage(int session_id, String message_type, String message_value) {
481    Log.info(String.format("message: [%d] [%s] %s", session_id, message_type, message_value));
482    Trime trime = Trime.getService();
483    switch (message_type) {
484      case "schema":
485        initSchema();
486        if (trime != null) {
487          trime.initKeyboard(false);
488          trime.updateComposing();
489        }
490        break;
491      case "option":
492        getStatus();
493        getContexts(); //切換中英文、簡繁體時更新候選
494        if (trime != null) {
495          trime.invalidateKeyboard(); //鍵盤狀態
496          if (message_value.endsWith("ascii_mode")) trime.setLanguage(isAsciiMode());
497        }
498        break;
499    }
500  }
