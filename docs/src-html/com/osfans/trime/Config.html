072  public static void prepareRime(Context context) {
073    boolean isExist = new File(USER_DATA_DIR).exists();
074    boolean isOverwrite = Function.isDiffVer(context);
075    if (isOverwrite) {
076      copyFileOrDir(context, RIME, true);
077    } else if (isExist) {
078      copyFileOrDir(context, RIME + "/" + defaultFile, false);
079    } else {
080      copyFileOrDir(context, RIME, false);
081    }
082    Rime.get(!isExist); //覆蓋時不強制部署
083  }

107  public static boolean deployOpencc() {
108    File d = new File(OPENCC_DATA_DIR);
109    if (d.exists()) {
110      FilenameFilter txtFilter = new FilenameFilter(){
111        @Override
112        public boolean accept(File dir, String filename) {
113          return filename.endsWith(".txt");
114        }
115      };
116      for (String txtName: d.list(txtFilter)) {
117        txtName = OPENCC_DATA_DIR + txtName;
118        String ocdName = txtName.replace(".txt", ".ocd");
119        Rime.opencc_convert_dictionary(txtName, ocdName, "text", "ocd");
120      }
121    }
122    return true;
123  }

136  public static boolean copyFileOrDir(Context context, String path, boolean overwrite) {
137    AssetManager assetManager = context.getAssets();
138    String assets[] = null;
139    try {
140      assets = assetManager.list(path);
141      if (assets.length == 0) {
142        copyFile(context, path, overwrite);
143      } else {
144        String fullPath = SDCARD + path;
145        File dir = new File(fullPath);
146        if (!dir.exists()) dir.mkdir();
147        for (int i = 0; i < assets.length; ++i) {
148          copyFileOrDir(context, path + "/" + assets[i], overwrite);
149        }
150      }
151    } catch (IOException ex) {
152      Log.e("Config", "I/O Exception", ex);
153      return false;
154    }
155    return true;
156  }

158  public static boolean copyFile(Context context, String filename, boolean overwrite) {
159    AssetManager assetManager = context.getAssets();
160    InputStream in = null;
161    OutputStream out = null;
162    try {
163      in = assetManager.open(filename);
164      String newFileName = SDCARD + filename;
165      if (new File(newFileName).exists() && !overwrite) return true;
166      out = new FileOutputStream(newFileName);
167      byte[] buffer = new byte[BLK_SIZE];
168      int read;
169      while ((read = in.read(buffer)) != -1) {
170          out.write(buffer, 0, read);
171      }
172      in.close();
173      in = null;
174      out.flush();
175      out.close();
176      out = null;
177    } catch (Exception e) {
178      Log.e("Config", e.getMessage());
179      return false;
180    }
181    return true;
182  }

194  public void init() {
195    String name = Rime.config_get_string(themeName, "config_version");
196    if (Function.isEmpty(name)) themeName = defaultName;
197    deployConfig();
198    mDefaultStyle = (Map<String,Object>)Rime.config_get_map(themeName, "style");
199    fallbackColors = (Map<String,String>)Rime.config_get_map(themeName, "fallback_colors");
200    List androidKeys = Rime.config_get_list(themeName, "android_keys/name");
201    Key.androidKeys = new ArrayList<String>(androidKeys.size());
202    for (Object o : androidKeys) {
203      Key.androidKeys.add(o.toString());
204    }
205    Key.presetKeys = (Map<String, Map>)Rime.config_get_map(themeName, "preset_keys");
206    presetColorSchemes = Rime.config_get_map(themeName, "preset_color_schemes");
207    presetKeyboards = Rime.config_get_map(themeName, "preset_keyboards");
208    reset();
209  }

216  private Object _getValue(String k1, String k2) {
217    Map<String, Object> m;
218    if (mStyle != null && mStyle.containsKey(k1)) {
219      m = (Map<String, Object>)mStyle.get(k1);
220      if (m != null && m.containsKey(k2)) return m.get(k2);
221    }
222    if (mDefaultStyle != null && mDefaultStyle.containsKey(k1)) {
223      m = (Map<String, Object>)mDefaultStyle.get(k1);
224      if (m != null && m.containsKey(k2)) return m.get(k2);
225    }
226    return null;
227  }
228
229  private Object _getValue(String k1) {
230    if (mStyle != null && mStyle.containsKey(k1)) return mStyle.get(k1);
231    if (mDefaultStyle != null && mDefaultStyle.containsKey(k1)) return mDefaultStyle.get(k1);
232    return null;
233  }

235  public Object getValue(String s) {
236    String[] ss = s.split("/");
237    if (ss.length == 1) return _getValue(ss[0]);
238    else if(ss.length == 2) return _getValue(ss[0], ss[1]);
239    return null;
240  }

246  public String getKeyboardName(String name) {
247    if (name.contentEquals(".default")) {
248      if (presetKeyboards.containsKey(schema_id)) name = schema_id; //匹配方案名
249      else {
250        if (schema_id.indexOf("_") >= 0) name = schema_id.split("_")[0];
251        if (!presetKeyboards.containsKey(name)) { //匹配“_”前的方案名
252          Object o = Rime.schema_get_value(schema_id, "speller/alphabet");
253          name = "qwerty"; //26
254          if (o != null) {
255            String alphabet = o.toString();
256            if (presetKeyboards.containsKey(alphabet)) name = alphabet; //匹配字母表
257            else {
258              if (alphabet.indexOf(",") >= 0 || alphabet.indexOf(";") >= 0) name += "_";
259              if (alphabet.indexOf("0") >= 0 || alphabet.indexOf("1") >= 0) name += "0";
260            }
261          }
262        }
263      }
264    }
265    if (!presetKeyboards.containsKey(name)) name = "default";
266    Map<String, Object> m = (Map<String, Object>)presetKeyboards.get(name);
267    if (m.containsKey("import_preset")) {
268      name = m.get("import_preset").toString();
269    }
270    return name;
271  }

273  public List<String> getKeyboardNames() {
274    List<String> names = (List<String>)getValue("keyboards");
275    List<String> keyboards = new ArrayList<String>();
276    for (String s: names) {
277      s = getKeyboardName(s);
278      if (!keyboards.contains(s)) keyboards.add(s);
279    }
280    return keyboards;
281  }

341  private Object getColorObject(String key) {
342    String scheme = getString("color_scheme");
343    Map map = (Map<String, Object>)presetColorSchemes.get(scheme);
344    Object o = map.get(key);
345    String fallbackKey = key;
346    while (o == null && fallbackColors.containsKey(fallbackKey)) {
347      fallbackKey = fallbackColors.get(fallbackKey);
348      o = map.get(fallbackKey);
349    }
350    return o;
351  }

404  public Drawable drawableObject(Object o) {
405    if (o == null) return null;
406    Integer color = null;
407    if (o instanceof Integer) color = ((Integer)o).intValue();
408    else if (o instanceof Float || o instanceof Double || o instanceof Long) color = ((Long)o).intValue();
409    else if (o instanceof String) {
410      String name = o.toString();
411      name = USER_DATA_DIR + "/backgrounds/" + name;
412      File f = new File(name);
413      if (f.exists()) {
414        return new BitmapDrawable(BitmapFactory.decodeFile(name));
415      }
416    }
417    if (color != null) {
418      GradientDrawable gd = new GradientDrawable();
419      gd.setColor(color);
420      return gd;
421    }
422    return null;
423  }

443  public int getInlinePreedit() {
444    if (getString("inline_code").contentEquals("true")) return INLINE_INPUT;
445    switch (getString("inline_preedit")) {
446        case "preview":
447        case "preedit":
448        case "true":
449          return INLINE_PREVIEW;
450        case "composition":
451          return INLINE_COMPOSITION;
452        case "input":
453          return INLINE_INPUT;
454    }
455    return INLINE_NONE;
456  }

459    WinPos wp = WinPos.fromString(getString("layout/position"));
