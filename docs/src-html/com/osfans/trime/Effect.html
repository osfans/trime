017package com.osfans.trime;
018
019import android.content.Context;
020import android.media.AudioManager;
021import android.os.Vibrator;
022import android.view.KeyEvent;
023import android.speech.tts.TextToSpeech;
024import android.os.Build.VERSION_CODES;
025import android.os.Build.VERSION;
026
027import java.util.Locale;
028
029/** 處理按鍵聲音、震動、朗讀等效果 */
030public class Effect {
031  private int duration = 30;
032  private float volume = -1.0f;
033
034  private final Context context;
035
036  private Vibrator vibrator;
037  private AudioManager audioManager;
038  private boolean isSpeakCommit, isSpeakKey;
039  private TextToSpeech mTTS;
040
041  public Effect(Context context) {
042    this.context = context;
043  }
044
045  public void reset() {
046    Config config = Config.get();
047    duration = config.getInt("key_vibrate_duration");
048    if (duration > 0 && (vibrator == null)) {
049      vibrator =
050        (Vibrator) context.getSystemService(Context.VIBRATOR_SERVICE);
051    }
052
053    volume = config.getFloat("key_sound_volume");
054    if (volume > 0) {
055      if (audioManager == null) {
056        audioManager =
057          (AudioManager) context.getSystemService(Context.AUDIO_SERVICE);
058      }
059      int max = audioManager.getStreamMaxVolume(AudioManager.STREAM_MUSIC);
060      volume = max * volume;
061      audioManager.setStreamVolume(AudioManager.STREAM_SYSTEM , (int)(volume), 0);
062    }
063
064    isSpeakCommit = config.getBoolean("speak_commit");
065    isSpeakKey = config.getBoolean("speak_key");
066    if (mTTS == null && (isSpeakCommit || isSpeakKey)) {
067      mTTS = new TextToSpeech(context, new TextToSpeech.OnInitListener() {
068        public void onInit(int status) {
069          //初始化結果
070        }
071      });
072    }
073  }
074
075  public void vibrate() {
076    if (duration > 0 && (vibrator != null)) vibrator.vibrate(duration);
077  }
078
079  public void playSound(final int code) {
080    if (volume > 0 && (audioManager != null)) {
081      final int sound;
082      switch (code) {
083        case KeyEvent.KEYCODE_DEL:
084            sound = AudioManager.FX_KEYPRESS_DELETE;
085            break;
086        case KeyEvent.KEYCODE_ENTER:
087            sound = AudioManager.FX_KEYPRESS_RETURN;
088            break;
089        case KeyEvent.KEYCODE_SPACE:
090            sound = AudioManager.FX_KEYPRESS_SPACEBAR;
091            break;
092        default:
093            sound = AudioManager.FX_KEYPRESS_STANDARD;
094            break;
095      }
096      audioManager.playSoundEffect(sound, -1);
097    }
098  }
099
100  public void setLanguage(Locale loc) {
101    if (mTTS != null) mTTS.setLanguage(loc);
102  }
103
104  public void speak(CharSequence text) {
105    if (text!= null && mTTS != null) mTTS.speak(text.toString(), TextToSpeech.QUEUE_FLUSH, null);
106  }
107
108  public void speakCommit(CharSequence text) {
109    if (isSpeakCommit) speak(text);
110  }
111
112  public void speakKey(CharSequence text) {
113    if (isSpeakKey) speak(text);
114  }
115
116  public void speakKey(int code) {
117    if (code <= 0) return;
118    if (VERSION.SDK_INT < VERSION_CODES.HONEYCOMB_MR1) return;
119    String text = KeyEvent.keyCodeToString(code).replace("KEYCODE_","").replace("_", " ").toLowerCase(Locale.getDefault());
120    speakKey(text);
121  }
122
123  public void destory() {
124     if (mTTS != null) {
125       mTTS.stop();
126       mTTS.shutdown();
127       mTTS = null;
128     }
129  }
130}
