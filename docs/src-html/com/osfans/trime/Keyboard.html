108    public Keyboard(Context context) {
109      DisplayMetrics dm = context.getResources().getDisplayMetrics();
110      mDisplayWidth = dm.widthPixels;
111      mDisplayHeight = dm.heightPixels;
112      //Log.v(TAG, "keyboard's display metrics:" + dm);
113
114      Config config = Config.get();
115      mDefaultHorizontalGap = config.getPixel("horizontal_gap");
116      mDefaultVerticalGap = config.getPixel("vertical_gap");
117      mDefaultWidth = (int)(mDisplayWidth * config.getDouble("key_width") / 100);
118      mDefaultHeight = config.getPixel("key_height");
119      mProximityThreshold = (int) (mDefaultWidth * SEARCH_DISTANCE);
120      mProximityThreshold = mProximityThreshold * mProximityThreshold; // Square it for comparison
121
122      mKeys = new ArrayList<Key>();
123      mComposingKeys = new ArrayList<Key>();
124    }

126    /**
127     * <p>Creates a blank keyboard from the given resource file and populates it with the specified
128     * characters in left-to-right, top-to-bottom fashion, using the specified number of columns.
129     * </p>
130     * <p>If the specified number of columns is -1, then the keyboard will fit as many keys as
131     * possible in each row.</p>
132     * @param context the application or service context
133     * @param characters the list of characters to display on the keyboard. One key will be created
134     * for each character.
135     * @param columns the number of columns of keys to display. If this number is greater than the 
136     * number of keys that can fit in a row, it will be ignored. If this number is -1, the 
137     * keyboard will fit as many keys as possible in each row.
138     * @param horizontalPadding 按鍵水平間距
139     */
140    public Keyboard(Context context, CharSequence characters, int columns, int horizontalPadding) {
141        this(context);
142        int x = 0;
143        int y = 0;
144        int column = 0;
145        mTotalWidth = 0;
146
147        final int maxColumns = columns == -1 ? Integer.MAX_VALUE : columns;
148        for (int i = 0; i < characters.length(); i++) {
149            char c = characters.charAt(i);
150            if (column >= maxColumns 
151                    || x + mDefaultWidth + horizontalPadding > mDisplayWidth) {
152                x = 0;
153                y += mDefaultVerticalGap + mDefaultHeight;
154                column = 0;
155            }
156            final Key key = new Key(this);
157            key.x = x;
158            key.y = y;
159            key.width = mDefaultWidth;
160            key.height = mDefaultHeight;
161            key.gap = mDefaultHorizontalGap;
162            key.events[0] = new Event(this, String.valueOf(c));
163            column++;
164            x += key.width + key.gap;
165            mKeys.add(key);
166            if (x > mTotalWidth) {
167                mTotalWidth = x;
168            }
169        }
170        mTotalHeight = y + mDefaultHeight; 
171    }

225  public boolean hasModifier(int modifiers) {

233  public boolean toggleModifier(int mask) {
234    boolean value = !hasModifier(mask);
235    if (value) mMetaState |= mask;
236    else mMetaState &= ~mask;
237    return value;
238  }

240  public int getModifer() {
241    return mMetaState;
242  }

244  public boolean setModifier(int mask, boolean value) {
245    boolean b = hasModifier(mask);
246    if (b == value) return false;
247    if (value) mMetaState |= mask;
248    else mMetaState &= ~mask;
249    return true;
250  }

280    private void computeNearestNeighbors() {
281        // Round-up so we don't have any pixels outside the grid
282        mCellWidth = (getMinWidth() + GRID_WIDTH - 1) / GRID_WIDTH;
283        mCellHeight = (getHeight() + GRID_HEIGHT - 1) / GRID_HEIGHT;
284        mGridNeighbors = new int[GRID_SIZE][];
285        int[] indices = new int[mKeys.size()];
286        final int gridWidth = GRID_WIDTH * mCellWidth;
287        final int gridHeight = GRID_HEIGHT * mCellHeight;
288        for (int x = 0; x < gridWidth; x += mCellWidth) {
289            for (int y = 0; y < gridHeight; y += mCellHeight) {
290                int count = 0;
291                for (int i = 0; i < mKeys.size(); i++) {
292                    final Key key = mKeys.get(i);
293                    if (key.squaredDistanceFrom(x, y) < mProximityThreshold ||
294                            key.squaredDistanceFrom(x + mCellWidth - 1, y) < mProximityThreshold ||
295                            key.squaredDistanceFrom(x + mCellWidth - 1, y + mCellHeight - 1) 
296                                < mProximityThreshold ||
297                            key.squaredDistanceFrom(x, y + mCellHeight - 1) < mProximityThreshold ||
298                            key.isInside(x, y) ||
299                            key.isInside(x + mCellWidth - 1, y) ||
300                            key.isInside(x + mCellWidth - 1, y + mCellHeight - 1) ||
301                            key.isInside(x, y + mCellHeight - 1)) {
302                        indices[count++] = i;
303                    }
304                }
305                int [] cell = new int[count];
306                System.arraycopy(indices, 0, cell, 0, count);
307                mGridNeighbors[(y / mCellHeight) * GRID_WIDTH + (x / mCellWidth)] = cell;
308            }
309        }
310    }
311    
312    /**
313     * Returns the indices of the keys that are closest to the given point.
314     * @param x the x-coordinate of the point
315     * @param y the y-coordinate of the point
316     * @return the array of integer indices for the nearest keys to the given point. If the given
317     * point is out of range, then an array of size zero is returned.
318     */
319    public int[] getNearestKeys(int x, int y) {
320        if (mGridNeighbors == null) computeNearestNeighbors();
321        if (x >= 0 && x < getMinWidth() && y >= 0 && y < getHeight()) {
322            int index = (y / mCellHeight) * GRID_WIDTH + (x / mCellWidth);
323            if (index < GRID_SIZE) {
324                return mGridNeighbors[index];
325            }
326        }
327        return new int[0];
328    }

330  public Keyboard(Context context, String name) {
331    this(context);
332    Map<String,Object> m = Config.get().getKeyboard(name);
333    mAsciiMode = (Integer)Key.getValue(m, "ascii_mode", 1);
334    int columns = (Integer)Key.getValue(m, "columns", 20);
335    int defaultWidth = (int)(Key.getDouble(m, "width", 0) * mDisplayWidth / 100);
336    if (defaultWidth == 0) defaultWidth = mDefaultWidth;
337    double height = Key.getDouble(m, "height", 0);
338    int defaultHeight = mDefaultHeight;
339    if (height > 0) defaultHeight = (int)TypedValue.applyDimension(TypedValue.COMPLEX_UNIT_SP, (float)height, Resources.getSystem().getDisplayMetrics());
340    int rowHeight = defaultHeight;
341    List<Map<String,Object>> lm = (List<Map<String,Object>>)m.get("keys");
342
343    int h_gap = m.containsKey("horizontal_gap") ? Key.getPixel(m, "horizontal_gap") : mDefaultHorizontalGap;
344    int v_gap = m.containsKey("vertical_gap") ? Key.getPixel(m, "vertical_gap") : mDefaultVerticalGap;
345    int x = h_gap/2;
346    int y = v_gap;
347    int row = 0;
348    int column = 0;
349    mTotalWidth = 0;
350
351    final int maxColumns = columns == -1 ? Integer.MAX_VALUE : columns;
352    for (Map<String,Object> mk: lm) {
353      int gap = h_gap;
354      int w = (int)(Key.getDouble(mk, "width", 0) * mDisplayWidth / 100);
355      if (w == 0 && mk.containsKey("click")) w = defaultWidth;
356      w -= gap;
357      if (column >= maxColumns || x + w > mDisplayWidth) {
358        x = gap/2;
359        y += v_gap + rowHeight;
360        column = 0;
361        row++;
362        if (mKeys.size() > 0) mKeys.get(mKeys.size() - 1).edgeFlags |= Keyboard.EDGE_RIGHT;
363      }
364      if (column == 0) {
365        double heightK = Key.getDouble(mk, "height", 0);
366        if (heightK > 0) {
367          rowHeight = (int)TypedValue.applyDimension(TypedValue.COMPLEX_UNIT_SP, (float)heightK, Resources.getSystem().getDisplayMetrics());
368        } else {
369          rowHeight = defaultHeight;
370        }
371      }
372      if (!mk.containsKey("click")){ //無按鍵事件
373        x += w + gap;
374        continue; //縮進
375      }
376
377      final Key key = new Key(this, mk);
378      key.x = x;
379      key.y = y;
380      int right_gap = Math.abs(mDisplayWidth - x - w - gap/2);
381      key.width = (right_gap <= mDisplayWidth / 100) ? mDisplayWidth - x - gap/2: w; //右側不留白
382      key.height = rowHeight;
383      key.gap = gap;
384      key.row = row;
385      key.column = column;
386      column++;
387      x += key.width + key.gap;
388      mKeys.add(key);
389      if (x > mTotalWidth) {
390          mTotalWidth = x;
391      }
392    }
393    if (mKeys.size() > 0) mKeys.get(mKeys.size() - 1).edgeFlags |= Keyboard.EDGE_RIGHT;
394    mTotalHeight = y + rowHeight + v_gap;
395    for (Key key: mKeys) {
396      if (key.column == 0) key.edgeFlags |= Keyboard.EDGE_LEFT;
397      if (key.row == 0) key.edgeFlags |= Keyboard.EDGE_TOP;
398      if (key.row == row) key.edgeFlags |= Keyboard.EDGE_BOTTOM;
399    }
400  }
