140  /**
141   * 選取候選項
142   *
143   * @param index 候選項序號（從0開始），{@code -1}表示選擇當前高亮候選項
144   * @return 是否成功選字
145   */
146  public boolean pickHighlighted(int index) {
147    if ((highlightIndex != -1) && (listener != null)) {
148      if (index == -1) index = highlightIndex;
149      if (index >= 0) index += start_num;
150      listener.onPickCandidate(index);
151      return true;
152    }
153    return false;
154  }

156  private boolean updateHighlight(int x, int y) {
157    int index = getCandidateIndex(x, y);
158    if (index != -1) {
159      highlightIndex = index;
160      invalidate();
161      return true;
162    }
163    return false;
164  }

166  private void removeHighlight() {
167    highlightIndex = -1;
168    invalidate();
169    requestLayout();
170  }

172  private boolean isHighlighted(int i) {
173    return candidate_use_cursor && i >=0 && i == highlightIndex;
174  }

176  private void drawHighlight(Canvas canvas) {
177    if (isHighlighted(highlightIndex)) {
178      candidateHighlight.setBounds(candidateRect[highlightIndex]);
179      candidateHighlight.draw(canvas);
180    }
181  }

183  private Typeface getFont(int codepoint, Typeface font) {
184    if (tfHanB != Typeface.DEFAULT && Character.isSupplementaryCodePoint(codepoint)) return tfHanB;
185    if (tfLatin != Typeface.DEFAULT && codepoint < 0x2e80) return tfLatin;
186    return font;
187  }
188
189  private void drawText(String s, Canvas canvas, Paint paint, Typeface font, float center, float y) {
190    if (s == null) return;
191    int length = s.length();
192    if (length == 0) return;
193    int points = s.codePointCount(0, length);
194    float x = center - measureText(s, paint, font) / 2;
195    if (tfLatin != Typeface.DEFAULT || (tfHanB != Typeface.DEFAULT && length > points)) {
196      int offset = 0;
197      while (offset < length) {
198        int codepoint = s.codePointAt(offset);
199        int charCount = Character.charCount(codepoint);
200        int end = offset + charCount;
201        paint.setTypeface(getFont(codepoint, font));
202        canvas.drawText(s, offset, end, x, y, paint);
203        x += paint.measureText(s, offset, end);
204        offset = end;
205      }
206    } else {
207      paint.setTypeface(font);
208      canvas.drawText(s, x, y, paint);
209    }
210  }

212  private void drawCandidates(Canvas canvas) {
213    if (candidates == null) return;
214
215    float x = 0;
216    float y = 0;
217    int i = 0;
218    float comment_x, comment_y;
219    float comment_width;
220    String candidate, comment;
221
222    y = candidateRect[0].centerY() - (paintCandidate.ascent() + paintCandidate.descent()) / 2;
223    if (show_comment && comment_on_top) y += comment_height / 2;
224    comment_y = comment_height / 2 - (paintComment.ascent() + paintComment.descent()) / 2;
225    if (show_comment && !comment_on_top) comment_y += candidateRect[0].bottom - comment_height;
226
227    while (i < num_candidates) {
228      // Calculate a position where the text could be centered in the rectangle.
229      x = candidateRect[i].centerX();
230      if (show_comment) {
231        comment = getComment(i);
232        if (!Function.isEmpty(comment)) {
233          comment_width = measureText(comment, paintComment, tfComment);
234          if (comment_on_top) {
235            comment_x = candidateRect[i].centerX();
236          } else {
237            x -= comment_width / 2;
238            comment_x = candidateRect[i].right -  comment_width / 2;
239          }
240          paintComment.setColor(isHighlighted(i) ? hilited_comment_text_color : comment_text_color);
241          drawText(comment, canvas, paintComment, tfComment, comment_x, comment_y);
242        }
243      }
244      paintCandidate.setColor(isHighlighted(i) ? hilited_candidate_text_color : candidate_text_color);
245      drawText(getCandidate(i), canvas, paintCandidate, tfCandidate, x, y);
246      // Draw the separator at the right edge of each candidate.
247      candidateSeparator.setBounds(
248        candidateRect[i].right - candidateSeparator.getIntrinsicWidth(),
249        candidateRect[i].top,
250        candidateRect[i].right + candidate_spacing,
251        candidateRect[i].bottom);
252      candidateSeparator.draw(canvas);
253      i++;
254    }
255    for (int j = -4; j >= -5; j--) { // -4: left, -5: right
256      candidate = getCandidate(j);
257      if (candidate == null) continue;
258      paintSymbol.setColor(isHighlighted(i) ? hilited_comment_text_color : comment_text_color);
259      x = candidateRect[i].centerX() - measureText(candidate, paintSymbol, tfSymbol) / 2;
260      canvas.drawText(candidate, x, y, paintSymbol);
261      candidateSeparator.setBounds(
262        candidateRect[i].right - candidateSeparator.getIntrinsicWidth(),
263        candidateRect[i].top,
264        candidateRect[i].right + candidate_spacing,
265        candidateRect[i].bottom);
266      candidateSeparator.draw(canvas);
267      i++;
268    }
269  }

271  @Override
272  protected void onDraw(Canvas canvas) {
273    if (canvas == null) {
274      return;
275    }
276    super.onDraw(canvas);
277
278    drawHighlight(canvas);
279    drawCandidates(canvas);

282  private void updateCandidateWidth() {
283    final int top = 0;
284    final int bottom = getHeight();
285    int i = 0;
286    int x = 0;
287    if (Rime.hasLeft()) x += getCandidateWidth(-4) + candidate_spacing;
288    getCandNum();
289    for (i = 0; i < num_candidates; i++) {
290      candidateRect[i] = new Rect(x, top, x += getCandidateWidth(i), bottom);
291      x += candidate_spacing;
292    }
293    if (Rime.hasLeft()) candidateRect[i++] = new Rect(0, top, (int)getCandidateWidth(-4), bottom);
294    if (Rime.hasRight()) candidateRect[i++] = new Rect(x, top, x += getCandidateWidth(-5), bottom);
295    LayoutParams params = getLayoutParams();
296    params.width = x;
297    params.height = candidate_view_height;
298    if (show_comment && comment_on_top) params.height += comment_height;
299    setLayoutParams(params);
300  }

313  @Override
314  public boolean onTouchEvent(MotionEvent me) {
315    int action = me.getAction();
316    int x = (int) me.getX();
317    int y = (int) me.getY();
318
319    switch (action) {
320      case MotionEvent.ACTION_DOWN:
321      case MotionEvent.ACTION_MOVE:
322        updateHighlight(x, y);
323        break;
324      case MotionEvent.ACTION_UP:
325        if (updateHighlight(x, y)) {
326          pickHighlighted(-1);
327        }
328        break;
329    }
330    performClick();
331    return true;
332  }

334  /**
335   * 獲得觸摸處候選項序號
336   * @param x 觸摸點橫座標
337   * @param y 觸摸點縱座標
338   * @return
339   * {@code >=0}: 觸摸點 (x, y) 處候選項序號，從0開始編號；
340   * {@code -1}: 觸摸點 (x, y) 處無候選項；
341   * {@code -4}: 觸摸點 (x, y) 處爲{@code Page_Up}；
342   * {@code -5}: 觸摸點 (x, y) 處爲{@code Page_Down}
343   */
344  private int getCandidateIndex(int x, int y) {
345    Rect r = new Rect();
346
347    int j = 0;
348    for (int i = 0; i < num_candidates; i++) {
349      // Enlarge the rectangle to be more responsive to user clicks.
350      r.set(candidateRect[j++]);
351      r.inset(0, CANDIDATE_TOUCH_OFFSET);
352      if (r.contains(x, y)) {
353        // Returns -1 if there is no candidate in the hitting rectangle.
354        return (i < num_candidates) ? i : -1;
355      }
356    }
357
358    if (Rime.hasLeft()) { //Page Up
359      r.set(candidateRect[j++]);
360      r.inset(0, CANDIDATE_TOUCH_OFFSET);
361      if (r.contains(x, y)) {
362        return -4;
363      }
364    }
365
366    if (Rime.hasRight()) { //Page Down
367      r.set(candidateRect[j++]);
368      r.inset(0, CANDIDATE_TOUCH_OFFSET);
369      if (r.contains(x, y)) {
370        return -5;
371      }
372    }
373
374    return -1;
375  }

377  private int getCandNum() {
378    candidates = Rime.getCandidates();
379    highlightIndex = Rime.getCandHighlightIndex() - start_num;
380    num_candidates = candidates == null ? 0 : candidates.length - start_num;
381    return num_candidates;
382  }

384  private String getCandidate(int i) {
385    String s = null;
386    if (candidates != null && i >= 0) s = candidates[i + start_num].text;
387    else if (i == -4 && Rime.hasLeft()) s = "◀";
388    else if (i == -5 && Rime.hasRight()) s = "▶";
389    return s;
390  }
391
392  private String getComment(int i) {
393    String s = null;
394    if (candidates != null && i >= 0) s = candidates[i + start_num].comment;
395    return s;
396  }

398  private float measureText(String s, Paint paint, Typeface font) {
399    float x = 0;
400    if (s == null) return x;
401    int length = s.length();
402    if (length == 0) return x;
403    int points = s.codePointCount(0, length);
404    if (tfLatin != Typeface.DEFAULT || (tfHanB != Typeface.DEFAULT && length > points)) {
405      int offset = 0;
406      while (offset < length) {
407        int codepoint = s.codePointAt(offset);
408        int charCount = Character.charCount(codepoint);
409        int end = offset + charCount;
410        paint.setTypeface(getFont(codepoint, font));
411        x += paint.measureText(s, offset, end);
412        offset = end;
413      }
414      paint.setTypeface(font);
415    } else {
416      paint.setTypeface(font);
417      x += paint.measureText(s);
418    }
419    return x;
420  }

422  private float getCandidateWidth(int i) {
423    String s = getCandidate(i);
424    float n = (s == null ? 0 : s.codePointCount(0, s.length()));
425    float x = 2 * candidate_padding;
426    if (s != null) x += measureText(s, paintCandidate, tfCandidate);
427    if (i >= 0 && show_comment) {
428      String comment = getComment(i);
429      if (comment != null) {
430        float x2 = measureText(comment, paintComment, tfComment);
431        if (comment_on_top) { if (x2 > x) x = x2; } //提示在上方
432        else x += x2;  //提示在右方
433      }
434    }
435    return x;
436  }
